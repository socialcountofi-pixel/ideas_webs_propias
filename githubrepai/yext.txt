// Variable base limpia
const API_BASE ="https://pokeapi.co/api/v2/";

// 1. Cargar las opciones en el SELECT al iniciar
async function initGrowthSelect() {
    const select = document.getElementById('growthSelect');
    
    try {
        const response = await fetch(`${API_BASE}/growth-rate`);
        const data = await response.json();
        
        // Limpiamos el select y ponemos las opciones reales
        select.innerHTML = '<option value="">-- Elige un ritmo --</option>';
        data.results.forEach(rate => {
            const option = document.createElement('option');
            option.value = rate.name; // Ejemplo: "slow", "fast"
            option.textContent = rate.name.toUpperCase();
            select.appendChild(option);
        });
    } catch (error) {
        console.error("Error al cargar ritmos:", error);
    }
}

// 2. Mostrar la lista cuando cambies el SELECT
async function showPokemonByGrowth(growthName) {
    if (!growthName) return;
    
    const listDiv = document.getElementById('pokemonList');
    listDiv.innerHTML = "Cargando Pokémon...";

    try {
        const response = await fetch(`${API_BASE}/growth-rate/${growthName}`);
        const data = await response.json();
        
        // Limpiamos y mostramos la lista de pokemon_species
        listDiv.innerHTML = ""; 
        data.pokemon_species.forEach(p => {
            const span = document.createElement('span');
            span.textContent = p.name;
            span.style.cssText = "padding: 5px 10px; background: #eee; border-radius: 5px; border: 1px solid #ccc; font-size: 14px;";
            listDiv.appendChild(span);
        });
    } catch (error) {
        listDiv.innerHTML = "Error al cargar la lista.";
        console.error(error);
    }
}

// 3. EVENTOS (Escuchar el cambio del select)
document.getElementById('growthSelect').addEventListener('change', (e) => {
    showPokemonByGrowth(e.target.value);
});

    const siguienteNivel = data.levels.find(l => l.level === (nivelActual ? nivelActual.level + 1 : 1));

    if (nivelActual) {
        let texto = `Nivel Actual: ${nivelActual.level}`;
        if (siguienteNivel) {
            const falta = siguienteNivel.experience - xpUser;
            texto += `<br>✨ ¡Te faltan ${falta.toLocaleString()} XP para el nivel ${siguienteNivel.level}!`;
        } else {
            texto += `<br>🏆 ¡Nivel Máximo alcanzado!`;
        }
        resultado.innerHTML = texto;

        resultado.innerHTML = "No tienes suficiente XP ni para el nivel 1.";
